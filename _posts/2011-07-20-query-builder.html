Function BuildQuery() {
    global $SelectValue;
    global $FromValue;
    global $WhereValue;
    global $OrderValue;

    $dbquery = $SelectValue;
    $dbquery .= $FromValue;
    $dbquery .= $WhereValue;
    if ($OrderValue) {
        $dbquery .= $OrderValue;
    }
    return($dbquery);
}
