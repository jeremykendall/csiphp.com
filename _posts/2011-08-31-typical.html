$subject = "Hi!";
$body = "Hi,\n\nHow are you?\n\n$_SERVER[REMOTE_ADDR]\n$hostname\n app=$app";
if (mail($to, $subject, $body)) {
  echo("
Message successfully sent!
");
 } else {
